class Solution(object):
    def moveZeroes(self, nums):
        init = 0
        prev = 0
        for i in range(len(nums)):
            if nums[i] == 0:
                init = init + 1
            if nums[i] != 0:
                hold = nums[i]
                nums[i] = nums[prev]
                nums[prev] = hold
                prev = prev + 1
                init = init + 1
        """
ok
        :type nums: List[int]
        :rtype: None Do not return anything, modify nums in-place instead.
        """
